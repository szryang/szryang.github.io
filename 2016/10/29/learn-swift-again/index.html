// 返回元组多个值
func calculateStatistics(scores: [Int]) -> (min: Int, max: Int, sum: Int) {
var min = scores[0]
var max = scores[0]
var sum = 0

for score in scores {
    if score < min {
        min = score
    } else if score > max {
        max = score
    }
    sum += score
}

return (min, max, sum)

}

let statistics = calculateStatistics(scores: [5, 3, 100, 3, 9])
print(statistics.min)
print(statistics.2)

// 函数可以带可变参数

sum += number
}
return sum

// 对象和类
class Shape {
var numberOfSides = 0
func simpleDescription() -> String {
return “A shape with (numberOfSides)”
}
}

var s = Shape()
print(s.simpleDescription())
s.numberOfSides = 9

class NamedShape {
var numberOfSides = 0
var name: String

init(name: String) {
    self.name = name
}

func simpleDescription() -> String {
    return "A shape with \(numberOfSides)  name -> \(name)"
}

}

var ns = NamedShape(name: “Kity”)
print(ns.simpleDescription())

// 继承
class Square : NamedShape {
var sideLength: Double = 0.0

init(sideLength: Double, name: String) {

    super.init(name: name)
    self.sideLength = sideLength
    numberOfSides = 4
}


func area() -> Double {
    return sideLength * sideLength
}

override func simpleDescription() -> String {
    return "A square with sides of length \(self.sideLength)"
}

}

var square = Square(sideLength: 14.4, name: “Square”)
print(“Square.area() -> (square.area())”)

// 扩展和协议
protocol ExampleProtocol {
var simpleDescription: String {get}
mutating func adjust()
}

class SimpleClass : ExampleProtocol {
internal func adjust() {
simpleDescription += “Now 100% adjusted..”
}

var simpleDescription: String = "A very simple class"

}

var sc = SimpleClass()
sc.adjust()
print(sc.simpleDescription)

extension Int {

func simpleDescription() -> String {

    return "Simple number is \(self)"
}

}

var snum: Int = 39
print(snum.simpleDescription())
